local O={}O.__index=O
local C={PRIMARY_RED=Color3.fromRGB(255,0,0),DARK_BLACK=Color3.fromRGB(20,20,20),LIGHT_BLACK=Color3.fromRGB(40,40,40),TEXT_WHITE=Color3.fromRGB(255,255,255),ACCENT_DARK_RED=Color3.fromRGB(200,0,0)}
local D={width=350,height=450,position=UDim2.new(0.5,-175,0.5,-225),draggable=true,showCloseButton=true,showMinimizeButton=true,title="O-X Panel"}
function O.new(t,c)local s=setmetatable({},O)s.config={}for k,v in pairs(D)do s.config[k]=c and c[k]or v end s.config.title=t or s.config.title s.isDragging=false s.dragStart=nil s.buttons={}s:_createMainFrame()s:_createHeader()s:_createContent()s:_setupDragFunctionality()return s end
function O:_createMainFrame()self.mainFrame=Instance.new("Frame")self.mainFrame.Name="OXPanel_Main"self.mainFrame.Size=UDim2.new(0,self.config.width,0,self.config.height)self.mainFrame.Position=self.config.position self.mainFrame.BackgroundColor3=C.DARK_BLACK self.mainFrame.BorderSizePixel=0 self.mainFrame.ZIndex=100 local b=Instance.new("UIStroke")b.Color=C.PRIMARY_RED b.Thickness=2 b.Parent=self.mainFrame local co=Instance.new("UICorner")co.CornerRadius=UDim.new(0,8)co.Parent=self.mainFrame local p=Instance.new("UIPadding")p.PaddingLeft=UDim.new(0,10)p.PaddingRight=UDim.new(0,10)p.PaddingTop=UDim.new(0,10)p.PaddingBottom=UDim.new(0,10)p.Parent=self.mainFrame end
function O:_createHeader()self.header=Instance.new("Frame")self.header.Name="Header"self.header.Size=UDim2.new(1,0,0,50)self.header.BackgroundColor3=C.LIGHT_BLACK self.header.BorderSizePixel=0 self.header.Parent=self.mainFrame local hc=Instance.new("UICorner")hc.CornerRadius=UDim.new(0,6)hc.Parent=self.header self.titleLabel=Instance.new("TextLabel")self.titleLabel.Name="Title"self.titleLabel.Size=UDim2.new(1,-100,1,0)self.titleLabel.Position=UDim2.new(0,10,0,0)self.titleLabel.BackgroundTransparency=1 self.titleLabel.TextColor3=C.TEXT_WHITE self.titleLabel.TextSize=18 self.titleLabel.Font=Enum.Font.GothamBold self.titleLabel.TextXAlignment=Enum.TextXAlignment.Left self.titleLabel.Text=self.config.title self.titleLabel.Parent=self.header if self.config.showMinimizeButton then self:_createHeaderButton("Minimize",60,function()self:toggleMinimize()end)end if self.config.showCloseButton then self:_createHeaderButton("X",30,function()self:destroy()end)end end
function O:_createHeaderButton(t,w,cb)local b=Instance.new("TextButton")b.Name=t.."Button"b.Size=UDim2.new(0,w,1,0)b.Position=UDim2.new(1,-w,0,0)b.BackgroundColor3=C.PRIMARY_RED b.TextColor3=C.TEXT_WHITE b.TextSize=16 b.Font=Enum.Font.GothamBold b.Text=t b.BorderSizePixel=0 b.Parent=self.header local co=Instance.new("UICorner")co.CornerRadius=UDim.new(0,4)co.Parent=b b.MouseEnter:Connect(function()b.BackgroundColor3=C.ACCENT_DARK_RED end)b.MouseLeave:Connect(function()b.BackgroundColor3=C.PRIMARY_RED end)b.MouseButton1Click:Connect(cb)b.TouchTap:Connect(cb)table.insert(self.buttons,b)end
function O:_createContent()self.contentFrame=Instance.new("Frame")self.contentFrame.Name="Content"self.contentFrame.Size=UDim2.new(1,0,1,-60)self.contentFrame.Position=UDim2.new(0,0,0,60)self.contentFrame.BackgroundColor3=C.DARK_BLACK self.contentFrame.BorderSizePixel=0 self.contentFrame.Parent=self.mainFrame local co=Instance.new("UICorner")co.CornerRadius=UDim.new(0,6)co.Parent=self.contentFrame local ll=Instance.new("UIListLayout")ll.Padding=UDim.new(0,8)ll.SortOrder=Enum.SortOrder.LayoutOrder ll.Parent=self.contentFrame local p=Instance.new("UIPadding")p.PaddingLeft=UDim.new(0,8)p.PaddingRight=UDim.new(0,8)p.PaddingTop=UDim.new(0,8)p.PaddingBottom=UDim.new(0,8)p.Parent=self.contentFrame end
function O:_setupDragFunctionality()if not self.config.draggable then return end local U=game:GetService("UserInputService")self.header.InputBegan:Connect(function(i,gp)if gp then return end if i.UserInputType==Enum.UserInputType.MouseButton1 then self.isDragging=true self.dragStart=i.Position end end)U.InputChanged:Connect(function(i,gp)if not self.isDragging or i.UserInputType~=Enum.UserInputType.MouseMovement then return end local d=i.Position-self.dragStart local np=self.mainFrame.AbsolutePosition+d self.mainFrame.Position=UDim2.new(0,np.X,0,np.Y)self.dragStart=i.Position end)U.InputEnded:Connect(function(i,gp)if i.UserInputType==Enum.UserInputType.MouseButton1 then self.isDragging=false end end)local ti=nil self.header.TouchBegan:Connect(function(t,gp)if gp then return end self.isDragging=true ti=t self.dragStart=t.Position end)self.header.TouchMoved:Connect(function(t,gp)if not self.isDragging or ti.UserInputType~=t.UserInputType then return end local d=t.Position-self.dragStart local np=self.mainFrame.AbsolutePosition+d self.mainFrame.Position=UDim2.new(0,np.X,0,np.Y)self.dragStart=t.Position end)self.header.TouchEnded:Connect(function(t,gp)self.isDragging=false ti=nil end)end
function O:toggleMinimize()self.isMinimized=not self.isMinimized if self.isMinimized then self.contentFrame.Visible=false self.mainFrame.Size=UDim2.new(0,self.config.width,0,50)else self.contentFrame.Visible=true self.mainFrame.Size=UDim2.new(0,self.config.width,0,self.config.height)end end
function O:addButton(t,cb)local b=Instance.new("TextButton")b.Name=t.."Btn"b.Size=UDim2.new(1,0,0,40)b.BackgroundColor3=C.PRIMARY_RED b.TextColor3=C.TEXT_WHITE b.TextSize=14 b.Font=Enum.Font.Gotham b.Text=t b.BorderSizePixel=0 b.Parent=self.contentFrame local co=Instance.new("UICorner")co.CornerRadius=UDim.new(0,4)co.Parent=b b.MouseEnter:Connect(function()b.BackgroundColor3=C.ACCENT_DARK_RED end)b.MouseLeave:Connect(function()b.BackgroundColor3=C.PRIMARY_RED end)b.MouseButton1Click:Connect(cb)b.TouchTap:Connect(cb)return b end
function O:addLabel(t)local l=Instance.new("TextLabel")l.Name="Label"l.Size=UDim2.new(1,0,0,30)l.BackgroundColor3=C.LIGHT_BLACK l.TextColor3=C.TEXT_WHITE l.TextSize=12 l.Font=Enum.Font.Gotham l.Text=t l.BorderSizePixel=0 l.Parent=self.contentFrame local co=Instance.new("UICorner")co.CornerRadius=UDim.new(0,4)co.Parent=l return l end
function O:addTextBox(p,cb)local ib=Instance.new("TextBox")ib.Name="TextInput"ib.Size=UDim2.new(1,0,0,35)ib.BackgroundColor3=C.LIGHT_BLACK ib.TextColor3=C.TEXT_WHITE ib.PlaceholderText=p or"Enter text..."ib.PlaceholderColor3=Color3.fromRGB(150,150,150)ib.TextSize=12 ib.Font=Enum.Font.Gotham ib.BorderSizePixel=0 ib.Parent=self.contentFrame local co=Instance.new("UICorner")co.CornerRadius=UDim.new(0,4)co.Parent=ib if cb then ib.FocusLost:Connect(function(ep)cb(ib.Text,ep)end)end return ib end
function O:setTitle(nt)if self.titleLabel then self.titleLabel.Text=nt self.config.title=nt end end
function O:setSize(w,h)self.config.width=w self.config.height=h if not self.isMinimized then self.mainFrame.Size=UDim2.new(0,w,0,h)end end
function O:setPosition(x,y)self.mainFrame.Position=UDim2.new(0,x,0,y)end
function O:toggleVisibility()self.mainFrame.Visible=not self.mainFrame.Visible return self.mainFrame.Visible end
function O:setVisible(v)self.mainFrame.Visible=v end
function O:isVisible()return self.mainFrame.Visible end
function O:setParent(p)self.mainFrame.Parent=p end
function O:destroy()if self.mainFrame then self.mainFrame:Destroy()end end
function O:getMainFrame()return self.mainFrame end
